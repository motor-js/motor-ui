## Import


## Usage

import { Props } from 'docz'
import { Playground } from 'docz'
import Pie from '../../lib/components/React/Pie/Pie.jsx';
import Motor from '../../lib/components/React/Motor/Motor.jsx';
import { config } from '../config/config.js'

<Playground>
<Motor config={config}>  
  <Pie
    title='Pie Chart'
    cols={[
      { qField: 'country', qLabel: 'Country' },
      { qField: '=Count(winery)', qLabel: 'No of Wineries' }
    ]}
  />
</Motor>

</Playground>

<br />

<Playground>
<Motor config={config}>  
  <Pie
    size='small'
    title='Donut Chart'
    subTitle='Sub Title'
    cols={[
      { qField: 'province', qLabel: 'Province' },
      { qField: '=Count(winery)', qLabel: 'No of Wineries' }
    ]}
    innerRadius={55}
    cornerRadius={5}
    padAngle={0.03}
    showLabels="outside"
  />
</Motor>
</Playground>

## Theming


### Entry Point

```javascript
theme.pie
```

### Object

Example below is taken from the Motor Theme.

```javascript

pie: {
   /** Configuration object to connect to the Qlik Engine. Must include Qlik site URL and an App name */
  config: PropTypes.object,
  /** Pie label  */
  label: PropTypes.string,
  /** cols from Qlik Data Model to render in the Pie  */
  cols: PropTypes.array.isRequired,
  /** Calc condition for the chart  */
  calcCondition: PropTypes.object,
  /** Supress zeo vlaues in the the chart  */
  suppressZero: PropTypes.bool,
  /** Pie width */
  width: PropTypes.string,
  /** The height of the Pie */
  height: PropTypes.string,
  /** The amount of margin around the component */
  margin: PropTypes.string,
  /** Whether the KPI size should scale for tablet and mobile */
  responsive: PropTypes.bool,
  /** Size of the Pie */
  size: PropTypes.string,
  /** Color of the Labels on the Pie */
  fontColor: PropTypes.string,
  /** Border of the Pie Chart. Enter as CSS style. Set to "none" to hide border */
  chartBorder: PropTypes.string,
  /** Background Color of the chart */
  backgroundColor: PropTypes.string,
  /** RoundNum of the Pie */
  roundNum: PropTypes.bool,
  /** Title of the Pie */
  title: PropTypes.string,
  /** Sub Title of the Pie */
  subTitle: PropTypes.string,
  /** Legend of the Pie */
  showLegend: PropTypes.bool,
  /** Display as Donut chart */
  innerRadius: PropTypes.number,
  /** Display as Donut chart */
  cornerRadius: PropTypes.number,
  /** Display as Donut chart */
  padAngle: PropTypes.number,
  /** Error messgae to display when invalid dimension or measure */
  dataErrorMsg: PropTypes.string,
  },

```
