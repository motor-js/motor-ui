## Usage

Display tabular data from your application. Includes sorting, conditional colouring, horizontal scroll and plenty of theming options.

import { Props } from 'docz'
import { Playground } from 'docz'
import Filter from '../../lib/components/React/Filter/Filter.jsx'
import Table from '../../lib/components/React/Table/Table.jsx'
import Pie from '../../lib/components/React/Pie/Pie.jsx'
import Motor from '../../lib/components/React/Motor/Motor.jsx'
import { config } from '../config/config.js'

<Playground>
    <Motor 
   config={config}>  
    <Table
      interactiveSort={true}
      tableWidth={'800px'}
      size='small'
      columns={[
      {
        dimensions: [
          {
            qField: 'country',
            qLabel: 'Country',
            qNullSuppression: false,
            width: '10%',
          }
        ],
        measures: [
          {
            qField: '=count(country)',
            qNumFormat: {
              qFmt: '#,###',
            },
          },
        ],
      },
    ]}
  /> 
  <Pie
    size='small'
    title='Donut Chart'
    subTitle='Sub Title'
    cols={[
      { qField: 'province', qLabel: 'Province' },
      { qField: '=Count(winery)', qLabel: 'No of Wineries' }
    ]}
    innerRadius={55}
    cornerRadius={5}
    padAngle={0.03}
    showLabels="outside"
  />
  </Motor>
</Playground>
